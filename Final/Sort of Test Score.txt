#include <iostream>
#include <stdlib.h>
using namespace std;
struct student_{
    string name;
    int score;
    int math;
    int chinese;
};
int main() {
    //cin
    int length;
    cin >> length;
    student_ student[length];
    int tempa,tempb;
    for (int i = 0; i < length; i++){
        cin >> student[i].name;
        cin >> tempa >> tempb;
        student[i].math = tempa;
        student[i].chinese = tempb;
        student[i].score = tempa + tempb;
    }
    for (int i = 0; i < length; i++){
        for (int j = 0; j < length - 1; j++){
            if (student[j].score < student[j + 1].score){
                student_ temp = student[j];
                student[j] = student[j + 1];
                student[j + 1] = temp;
            }
        }
    }
    cout << "\n\n";
    for (int i = 0; i < length; i++){
        cout << student[i].name << " " <<student[i].math << " " << student[i].chinese << " " << student[i].score << "\n";
    }

  
}
